#include <Arduino.h>
#include <Wire.h>

/** * @brief Enables the backlight via the CH422G IO Expander 
 */
void initBacklight() {
    Wire.begin(8, 9);
    Wire.beginTransmission(0x24); // CH422G I2C Address
    Wire.write(0x01);             // Set output mode
    Wire.endTransmission();
}

void setup() {
    Serial.begin(115200);
    initBacklight();
    /* Initialize Arduino_GFX or LVGL here */
}
